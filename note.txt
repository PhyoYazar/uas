 ===> `
  SELECT
     s.student_id,
     s.student_name,
     co.co_id,
     co.name AS co_name, co.instance co_instance,
     SUM(fm.mark) AS total_full_marks,
     SUM(sm.mark) AS total_marks
 FROM
     students s
 JOIN
     student_marks sm ON s.student_id = sm.student_id
 JOIN
     attributes a ON sm.attribute_id = a.attribute_id
 JOIN
     co_attributes ca ON a.attribute_id = ca.attribute_id
 JOIN
     course_outlines co ON ca.co_id = co.co_id
 JOIN
     full_marks fm ON sm.subject_id = fm.subject_id AND sm.attribute_id = fm.attribute_id where s.academic_year = '2010-2011' and s.year = 'Second Year'
 GROUP BY
     s.student_id, s.student_name, co.co_id, co_name
 ORDER BY
     s.student_id, co.co_id;`
